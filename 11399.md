대기시간+처리시간을 가장 적게 만드는 문제였다.   
정답률이 66%인만큼 읽자마자 바로 처리시간이 짧은 사람부터 정렬하면 되겠다고 생각했다!

```java
public class Baekjoon11399 {
	public static void main(String[] args) throws NumberFormatException, IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		//입력받는 사람 수
		int n= Integer.parseInt(br.readLine());
		//사람들의 인출시간
		String arr = br.readLine();
		//문자열을 split으로 자르고 저장할 문자열 배열
		String pArr[] = new String[n];
		//인출시간의 합
		int sum=0;
		//인출시간 배열, 반복문의 인덱스를 1부터 시작하기 위해 크기를 +1 해준다
		int[] p = new int[n+1];
		
		//split()으로 " " 을 기준으로 배열에 저장
		pArr = arr.split(" ");
		
		//String형 배열 -> int형 배열 형변환
		for(int i=0; i<pArr.length; i++) {
			p[i] = Integer.parseInt(pArr[i]);
		}
		
		//sort() 메서드로 오름차순 정렬
		Arrays.sort(p);
		
		//인덱스를 1부터 시작해 p[i] = p[i-1] + p[i] 으로 i일때는 i 이전의 모든 값을 더한 값이 된다
		for(int i=1; i<=n; i++) {
			p[i] = p[i-1]+p[i];
			sum += p[i];
		}
		System.out.println(sum);
		
	}
}

```
